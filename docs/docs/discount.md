## Get discount list

To retrieve a list of all public discount, witch you can filter with parameters

```ts
const discountlist = await sherl.discount().list(1,10 {your_key: "Your_value"})
```

Return object

```ts
{
  "results": [
      {
        id: string;
        uri: string;
        name: string;
        ownerUri: string; 
        owner: IOrganizationResponse;
        consumerId: string;
        availableFrom: Date; 
        availableUntil: Date;
        public: boolean;
        visibleToPublic: boolean;
        enabled: boolean;
        highlight: boolean; 
        cumulative: boolean;
        discountType: DiscountTypeEnum; 
        code: string; 
        percentage: number;
        amount: number;
        quantity: number; 
        quantityPerUser: number; 
        customers: string[]; 
        productRestrictions: IProductRestriction[]; 
        dateRestrictions: IDateRestriction[]; 
        createdAt: Date;
        updatedAt: Date;
      }],
  "view": {
    "total": number_of_item,
    "page": "1",
    "itemsPerPage": "10"
  }
```

